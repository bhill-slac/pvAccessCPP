112 /** A smart pointer to extract a MonitorElement from a Monitor queue
113  *
114  * To fetch a single element
115  @code
116  epics::pvAccess::Monitor::shared_pointer mon(....);
117  epics::pvAccess::MonitorElement::Ref elem(mon);
118  if(elem) {
119  // do something with element
120  assert(elem->pvStructurePtr->getSubField("foo"));
121  } else {
122  // queue was empty
123  }
124  @endcode
125  * To fetch all available elements (c++11)
126  @code
127  epics::pvAccess::Monitor::shared_pointer mon(....);
128  for(auto& elem : *mon) {
129  assert(elem.pvStructurePtr->getSubField("foo"));
130  }
131  @endcode
132  * To fetch all available elements (c++98)
133  @code
134  epics::pvAccess::Monitor::shared_pointer mon(....);
135  for(epics::pvAccess::MonitorElement::Ref it(mon); it; ++it) {
136  MonitorElement& elem(*it);
137  assert(elem.pvStructurePtr->getSubField("foo"));
138  }
139  @endcode
140  */
141 class MonitorElement::Ref
142 {
143  Monitor* mon;
144  MonitorElementPtr elem;
145 public:
146  Ref() :mon(0), elem() {}
147  Ref(Monitor& M) :mon(&M), elem(mon->poll()) {}
148  Ref(const Monitor::shared_pointer& M) :mon(M.get()), elem(mon->poll()) {}
149  ~Ref() { reset(); }
150  void attach(Monitor& M) {
151  reset();
152  mon = &M;
153  }
154  void attach(const Monitor::shared_pointer& M) {
155  reset();
156  mon = M.get();
157  }
158  bool next() {
159  if(elem) mon->release(elem);
160  elem = mon->poll();
161  return !!elem;
162  }
163  void reset() {
164  if(elem && mon) mon->release(elem);
165  elem.reset();
166  }
167  Ref& operator++() {// prefix increment. aka "++(*this)"
168  next();
169  return *this;
170  }
171 #if __cplusplus>=201103L
172  inline explicit operator bool() const { return elem.get(); }
173 #else
174 private:
175  typedef const Monitor* const * hidden_bool_type;
176 public:
177  operator hidden_bool_type() const { return elem.get() ? &mon : 0; }
178 #endif
179  inline MonitorElement* operator->() { return elem.get(); }
180  inline MonitorElement& operator*() { return *elem; }
181  inline MonitorElement* get() { return elem.get(); }
182 
183  inline bool operator==(const Ref& o) const { return elem==o.elem; }
184  inline bool operator!=(const Ref& o) const { return !(*this==o); }
185 };
